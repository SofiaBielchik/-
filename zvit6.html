                    function getProductDetails(productId, successCallback, errorCallback) {
                        
                        const products = {
                            '123': { name: 'Хліб', price: 50 },
                            '456': { name: 'Майонез', price: 75 },
                            '987': { name: 'Риба', price: 130 },
                        };
            
                        if (products[productId]) {
                            
                            successCallback(products[productId]);
                        } else {
                           
                            errorCallback('Продукту за таким ID не знайдено');
                        }
                    }
            
                    
                    function onSuccess(details) {
                        const resultContainer = document.getElementById('resultContainer');
                        resultContainer.innerHTML = `
Назва продукту: ${details.name}

                                                     
Ціна: ${details.price}
`;
                    }
            
                    
                    function onError(message) {
                        const resultContainer = document.getElementById('resultContainer');
                        resultContainer.innerHTML = 
Помилка: ${message}
;
                    }
            
                    
                    function searchProduct() {
                        const productIdInput = document.getElementById('productIdInput');
                        const productId = productIdInput.value;
            
                        getProductDetails(productId, onSuccess, onError);
                    }

                    // Функція для застосування знижки на медикаменти
                    function applyDiscount(medicines) {
                        return medicines.map(medicine => {
                            if (medicine.price > 300) {
                                return { ...medicine, discountedPrice: 0.7 * medicine.price };
                            } else {
                                return medicine;
                            }
                        });
                    }
            
                    // Вивід результату на сторінці

                    // Функція конструктора Storage
                    function Storage(initialItems) {
                        this.items = initialItems || [];
            
                        this.getItems = function () {
                            return this.items;
                        };
            
                        this.addItem = function (item) {
                            this.items.push(item);
                        };
            
                        this.removeItem = function (item) {
                            const index = this.items.indexOf(item);
                            if (index !== -1) {
                                this.items.splice(index, 1);
                            }
                        };
                    }
            
                    // Створення екземпляра Storage
                    const storage = new Storage(["яблуко", "банан", "манго"]);

                    // Функція для перевірки правильності закриття дужок
                    function checkBrackets(str) {
                        const stack = [];
                        const brackets = { '(': ')', '{': '}', '[': ']' };
            
                        for (let char of str) {
                            if (brackets[char]) {
                                // Якщо символ є відкриваючою дужкою, додаємо її в стек
                                stack.push(char);
                            } else if (Object.values(brackets).includes(char)) {
                                // Якщо символ є закриваючою дужкою
                                const lastBracket = stack.pop();
                                if (brackets[lastBracket] !== char) {
                                    // Перевіряємо, чи відповідає закриваюча дужка останній відкриваючій
                                    return false;
                                }
                            }
                        }
            
                        // Перевіряємо, чи всі відкриваючі дужки були закриті
                        return stack.length === 0;
                    }
            
                    // Приклади вхідних рядків для перевірки
                    const code1 = "function someFn() { console.log('Hello World'); }";
                    const code2 = "function someFn() { console.log('Hello World'; }";
            
                    // Виведення результатів на сторінці
                    document.write(
Код 1: ${checkBrackets(code1)}
);
                    document.write(
Код 2: ${checkBrackets(code2)}
);
